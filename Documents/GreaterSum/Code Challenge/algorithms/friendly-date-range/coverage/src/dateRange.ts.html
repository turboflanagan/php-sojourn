export class FriendlyDate {
    private _uglyDate: string[];
    private _dayLookup: any = {
        "01": "1st",
        "02": "2nd",
        "05": "5th",
        "07": "7th"
    };
    private _monthLookup: any = {
        "01": "January",
        "02": "February",
        "03": "March",
        "04": "April",
        "05": "May",
        "06": "June",
        "07": "July",
        "08": "August",
        "09": "September",
        "10": "October",
        "11": "November",
        "12": "December"
    };
 
    constructor(uglyDate: string[]) {
        this._uglyDate = uglyDate;
    };
 
    getPrettyDate(): string[] {
        let splitStartDate = this._uglyDate[0].split("-");
        let splitEndDate = this._uglyDate[1].split("-");
        if (this._uglyDate[0] === this._uglyDate[1]) {
            return [this.fullFriendlyDate(splitStartDate)];
        }
        let startDate = this.getStartDate(splitStartDate, splitEndDate);
        let endDate = this.getEndDate(splitEndDate, splitStartDate);
 
        return [startDate, endDate];
    }
 
    private fullFriendlyDate(splitStartDate: string[]) {
        return this._monthLookup[splitStartDate[1]] + " " + this._dayLookup[splitStartDate[2]] + ", " + splitStartDate[0];
    }
 
    private getStartDate(splitStartDate: string[], splitEndDate: string[]): string {
        let startYear: string = "";
        let startDayAndMonth: string = this._monthLookup[splitStartDate[1]] + " " + this._dayLookup[splitStartDate[2]];
        if (this.notCurrentYear(splitStartDate) || this.endYearGreaterThanStartYear(splitEndDate, splitStartDate)){
            startYear = ", " + splitStartDate[0];
        }
        return startDayAndMonth + startYear;
    }
 
    private getEndDate(splitEndDate: string[], splitStartDate: string[]) {
        let endYear: string = "";
        let endMonth: string = "";
        let endDay = this._dayLookup[splitEndDate[2]];
        if (this.monthsDifferent(splitEndDate, splitStartDate)){
            endMonth = this._monthLookup[splitEndDate[1]] + " ";
        }
        if (this.endYearGreaterThanStartYear(splitEndDate, splitStartDate)) {
            endMonth = this._monthLookup[splitEndDate[1]] + " ";
            endYear = ", " + splitEndDate[0];
        }
        if ((this.endYearIsOneMoreThanStartYear(splitEndDate, splitStartDate) && this.endMonthComesBeforeStartMonth(splitEndDate, splitStartDate))
            || (this.endYearIsOneMoreThanStartYear(splitEndDate, splitStartDate) && !this.monthsDifferent(splitEndDate, splitStartDate) && this.endIsLowerThanStartDay(splitEndDate, splitStartDate))) {
            endYear = "";
        }
 
        return endMonth + endDay + endYear;
    }
 
    private endsSameDay(splitEndDate: string[], splitStartDate: string[]) {
        return splitEndDate === splitStartDate;
    }
 
    private endIsLowerThanStartDay(splitEndDate: string[], splitStartDate: string[]) {
        return splitEndDate[2] < splitStartDate[2];
    }
 
    private endMonthComesBeforeStartMonth(splitEndDate: string[], splitStartDate: string[]) {
        return parseInt(splitEndDate[1]) < parseInt(splitStartDate[1]);
    }
 
    private endYearIsOneMoreThanStartYear(splitEndDate: string[], splitStartDate: string[]) {
        return parseInt(splitEndDate[0]) === parseInt(splitStartDate[0]) + 1;
    }
 
    private endYearGreaterThanStartYear(splitEndDate: string[], splitStartDate: string[]) {
        return splitEndDate[0] > splitStartDate[0];
    }
 
    private monthsDifferent(splitEndDate: string[], splitStartDate: string[]) {
        return splitEndDate[1] !== splitStartDate[1];
    }
 
    private notCurrentYear(splitStartDate: string[]) {
        return splitStartDate[0] !== "2016";
    }
}
